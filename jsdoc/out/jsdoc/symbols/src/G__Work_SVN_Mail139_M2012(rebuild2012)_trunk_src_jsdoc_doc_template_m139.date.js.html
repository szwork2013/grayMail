 15         *解析文本为日期对象，目前只支持yyyy-MM-dd hh:mm:ss的写法，解析失败返回null
 16         *@param {String|Number} 要解析的时间文本,"yyyy-MM-dd hh:mm:ss"格式，或者是秒数
 17         *@returns {Date}
 18         *@example
 19         $Date.parse("2012-10-10 10:10:10");
 20         */
 21         parse: function (str) {
 22             if (/^\d{10}$/.test(str)) {
 23                 return new Date(str * 1000);
 24             } else if (/^\d{13}$/.test(str)) {
 25                 return new Date(str * 1);
 26             }
 27 
 28             str = str.trim();
 29             var reg = /^(\d{4})-(\d{1,2})-(\d{1,2}) (\d{1,2}):(\d{1,2}):(\d{1,2})$/;
 30             var m = str.match(reg);
 31             if (m) {
 32                 var year = m[1];
 33                 var month = parseInt(m[2] - 1, 10);
 34                 var day = parseInt(m[3], 10);
 35                 var hour = parseInt(m[4], 10);
 36                 var minutes = parseInt(m[5], 10);
 37                 var seconds = parseInt(m[6], 10);
 38                 return new Date(year, month, day, hour, minutes, seconds);
 39             } else {
 40                 return null;
 41             }
 42         },
 43         /**
 44         *格式化时间文本
 45         *@param {Date} text 要格式化的文本
 46         *@param {String} date 时间对象
 47         *@returns {String}
 48         @example
 49         $Date.format("现在是yyyy年MM月dd日 hh点mm分ss秒，星期w",new Date());
 50         y 表示年份
 51         M 大写M表示月份
 52         d 表示几号
 53         h 表示小时
 54         m 表示分
 55         s 表示秒
 56         w 表示星期几
 57         */
 58         format: function (text, date) {
 59             var o = {
 60                 "M+": date.getMonth() + 1, //month
 61                 "d+": date.getDate(),    //day
 62                 "h+": date.getHours(),   //hour
 63                 "m+": date.getMinutes(), //minute
 64                 "s+": date.getSeconds(), //second
 65                 "q+": Math.floor((date.getMonth() + 3) / 3), //quarter
 66                 "S": date.getMilliseconds(), //millisecond
 67                 "w": "日一二三四五六".charAt(date.getDay())
 68             };
 69             text = text.replace(/y{4}/, date.getFullYear())
 70             .replace(/y{2}/, date.getFullYear().toString().substring(2))
 71             for (var k in o) {
 72                 var reg = new RegExp(k);
 73                 text = text.replace(reg, match);
 74             }
 75             function match(m) {
 76                 return m.length == 1 ? o[k] : ("00" + o[k]).substr(("" + o[k]).length);
 77             }
 78             return text;
 79         },

118         getFriendlyString: function (date, now) {
119             if (!date) return "";
120             if (typeof date == "number") date = new Date(date * 1000);
121             now = now || new Date();
122             var result;
123             //今天的邮件
124             var t = now.getTime() - date.getTime(); 	//相差毫秒
125             if (t < 0) {
126                 result = this.format("yyyy-M-dd", date);
127             }
128             else if (date.getYear() == now.getYear() && date.getMonth() == now.getMonth() && date.getDate() == now.getDate()) {
129                 var minutes = Math.round(t / 1000 / 60);
130                 if (minutes == 0) {
131                     result = "刚刚";
132                 } else if (minutes > 0 && minutes < 60) {
133                     result = minutes + "分钟前";
134                 } else {
135                     result = Math.floor(minutes / 60) + "小时前";
136                 }
137             } else if (date.getYear() == new Date().getYear()) {
138                 result = this.format("M-dd(w)", date);
139             } else {
140                 result = this.format("yyyy-M-dd(w)", date);
141             }
142             return result;
