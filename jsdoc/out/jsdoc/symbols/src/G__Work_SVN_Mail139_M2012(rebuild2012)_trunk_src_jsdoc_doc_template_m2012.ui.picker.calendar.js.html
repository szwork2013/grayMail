 82             if (this.stopPassDate) {
 83                 var prevYear = new Date(this.curValue);
 84                 prevYear.setFullYear(prevYear.getFullYear() - 1);
 85                 if (this.compareMonth(new Date,prevYear) > 0) {
 86                     return;

 99         /**
100          *@inner
101          */
102         onPrevMonthClick: function () {
103             if (this.stopPassDate && this.isCurrentMonth(this.curValue)) {
104                 //禁选过去月

106             }
107 
108             this.curValue.setDate(0);
109             this.updateContent(this.curValue);
110         },

112          *@inner
113          */
114         onNextMonthClick:function(){

116             this.updateContent(this.curValue);

123         updateContent: function (date) {
124             this.$("tbody").html(this.getCalendarHTML(date));
125             this.$(".MonthLabel").text(date.format("yyyy-MM"));
126             this.curValue = new Date(date);

131         /**

135         focusSelectedCell:function(){

142         onDateClick:function(e){
143             var td = e.target;
144             var date = td.innerHTML;
145             
146             if(/\d+/.test(date)){
147                 var selDate = new Date(this.curValue);
148                 selDate.setDate(date);
149

159                 this.onSelect(selDate);
160                 this.hide();
161             }

164         //是否过去的月份
165         compareMonth:function(date1,date2){
166             if (date1.getFullYear() > date2.getFullYear()) {
167                 return 1;
168             } else if (date1.getFullYear() < date2.getFullYear()) {
169                 return -1;

172             }

174

181         /**根据日期获得日期区域的html内容*/
182         getCalendarHTML:function(date){
183             var days = M139.Date.getDaysOfMonth(date);
184             var firstMonthDay = M139.Date.getFirstWeekDayOfMonth(date);
185             var htmlCode = [];
186             var cellsCount = days + firstMonthDay + 1;
187 
188             //是否禁选过去时间
189             var stopPassDate = this.stopPassDate;
190             var passMonth = this.compareMonth(new Date(),date);
191             var today = new Date().getDate();
192 
193             htmlCode.push("<tr>");
194  
195             for(var i=1,j=1;i<=cellsCount;i++,j++){
196                 if(i>firstMonthDay && j<=days){
197                     htmlCode.push("<td rel='" + j + "' " + getColor(j) + ">" + j + "</td>");
198                 }else{
199                     htmlCode.push("<td></td>");
200                     j--;
201                 }
202                 if(i%7 == 0 || i==cellsCount){
203                     htmlCode.push("</tr>");
204                 }
205             }
206             function getColor(date) {
207                 if (!stopPassDate) return "";
208                 var disableColor = 'style="color:silver;"';
209                 if (passMonth > 0) {
210                     return disableColor;
211                 } else if (passMonth < 0) {
212                     return "";
213                 }else{
214                     return date < today ? disableColor : "";
215                 }
216             }
217             return htmlCode.join("");
218         }
