 10     M139.namespace("M2012.UI.HTMLEditor.View.Menu", superClass.extend(
 11         /**
 12         *@lends M2012.UI.HTMLEditor.View.Menu.prototype
 13         */
 14         {
 15             /** 弹出菜单组件
 16             *@constructs M2012.UI.HTMLEditor.View.Menu
 17             *@extends M139.View.ViewBase
 18             *@param {Object} options 初始化参数集
 19             *@param {String} options.template 组件的html代码
 20             *@example
 21             */
 22             initialize: function (options) {
 23                 var $el = jQuery((options && options.template) || this.template);
 24                 this.setElement($el);
 25                 return superClass.prototype.initialize.apply(this, arguments);
 26             },
 27 
 28             render: function () {
 29                 var This = this;
 30                 this.$el.appendTo(document.body);
 31 
 32                 this.on("select", this.hide);
 33 
 34                 this.render = function () {
 35                     return this;
 36                 }
 37 
 38                 return superClass.prototype.render.apply(this, arguments);
 39             },
 40 
 41             hide: function () {
 42                 M2012.UI.PopMenu.unBindAutoHide({ action: "click", element: this.el})
 43                 return superClass.prototype.hide.apply(this, arguments);
 44             },
 45 
 46             //#xxx转rgb
 47             getRGBColor: function (color) {
 48                 if (/rgb/i.test(color)) {
 49                     return color.toLowerCase();
 50                 } else if (color.indexOf("#") > -1) {
 51                     var m = color.match(/^\#(.)(.)(.)$/);
 52                     if (m) {
 53                         return M139.Text.Utils.format("rgb({r},{g},{b})", {
 54                             r: parseInt(m[1] + m[1], 16),
 55                             g: parseInt(m[2] + m[2], 16),
 56                             b: parseInt(m[3] + m[3], 16)
 57                         });
 58                     } else {
 59                         m = color.match(/^\#(..)(..)(..)$/);
 60                         if (m) {
 61                             return M139.Text.Utils.format("rgb({r},{g},{b})", {
 62                                 r: parseInt(m[1], 16),
 63                                 g: parseInt(m[2], 16),
 64                                 b: parseInt(m[3], 16)
 65                             });
 66                         }
 67                     }
 68                 }
 69                 return color;
 70             },
 71 
 72             /**
 73              *显示菜单
 74              *@param {Object} options 参数集
 75              *@param {HTMLElement} options.dockElement 停靠的元素
 76              */
 77             show: function (options) {
 78                 var This = this;
 79                 this.$el.css("z-index", 40000);
 80                 this.dockElement = options.dockElement;
 81                 //停靠在按钮旁边
 82                 M139.Dom.dockElement(options.dockElement, this.el, {
 83                     direction:"down"
 84                 });
 85                 //点击空白处自动消失
 86                 M2012.UI.PopMenu.bindAutoHide({
 87                     action: "click",
 88                     element: this.el,
 89                     stopEvent: true,
 90                     callback: function () {
 91                         This.hide();
 92                     }
 93                 });
 94                 return superClass.prototype.show.apply(this, arguments);
 95             }
 96         }
