  9 /**
 10     *标准版框架基础类，实例化以后是M139.core (小写)
 11     *@class M139.Core
 12     *@memberOf M139
 13 */
 14 M139.Core = Backbone.Model.extend(
 15     /**
 16     *@lends M139.Core.prototype
 17     */
 18     {
 19     /**
 20     *JS模块异步加载状态，未加载|加载中|已加载
 21     *@inner
 22     */
 23     NSLOADEDSTATUS: {
 24         "NOTLOAD": "none",
 25         "LOADING": "loading",
 26         "LOADED": "loaded"
 27     },
 28     /**
 29     *定义异常
 30     *@inner
 31     */
 32     THROWS:{
 33         "NSNOTEXIST":"namespace not registered:"
 34     },
 35     /***@inner*/
 36     defaults: {
 37         /**
 38         *JS文件与模块对应的文件名存放数组
 39         */
 40         definedJSNSPath: {},
 41         loadingJSPath: {},
 42         loadedJSPath: {}
 43     },
 44     /**
 45     *构造函数
 46     *@constructs
 47     *@param {Object} options 初始化参数集，主要是一些配置信息
 48     *@param {Stirng} options.jsPath 声明JS文件存放的路径
 49     */
 50     initialize: function (options) {
 51         this.jsPath = options.jsPath;
 52     },
 53     /**
 54     *声明命名空间与对应的JS对象
 55     *@param {Stirng} namespace 名称空间，比如M139.dom
 56     *@param {Object} obj 名称空间对应的JS对象
 57     *@param {Window} win 注册名称空间的window对象
 58     * @example
 59     * M139.core.namespace("M139.dom",{ hasClass:function(){} });
 60     */
 61     namespace: function (namespace, obj, win) {
 62         var path = namespace.split(".");
 63         var target = win || window;
 64         while (path.length > 0) {
 65             var p = path.shift();
 66             if (!target[p]) {
 67                 if (path.length > 0) {
 68                     target[p] = {};
 69                 } else {
 70                     target[p] = obj;
 71                 }
 72             }
 73             target = target[p];
 74         }
 75         /**命名空间注册产生的事件
 76             * @name M139.Core#namespace
 77             * @event
 78             * @param {String} ns 加载成功的命名空间
 79             * @example M139.core.on("namespace",function(ns){});
 80         */
 81         this.trigger("namespace",namespace);
 82     },
 83     /**
 84     *@inner
 85     */
 86     getJSPathByNS: function (ns) {
 87         var container = this.get("definedJSNSPath");
 88         var p = container[ns];
 89         if(!p){
 90             throw this.THROWS.NSNOTEXIST;
 91         }
 92         return this.jsPath + "/" + p;
 93     },
 94     /**
 95     *声明名称空间与对应的JS文件路径
 96     *@param {String} namespace 要注册名称空间
 97     *@param {String} jsPath 实际JS文件路径
 98     * @example
 99     * M139.core.registerJS("M139.dom","m139.dom.js");
100     */
101     registerJS: function (namespace, jsPath) {
102         var container = this.get("definedJSNSPath");
103         container[namespace] = jsPath;
104         this.trigger("registerjs", { namespace: namespace, path: jsPath });
105     },
106     /**
107     *加载名称空间类库，如果已加载则不重复加载
108     *@param {Array} usingNSList 要引入的模块
109     *@param {Function} callback 模块引入后一步调用callback
110     * @example
111     * M139.core.requireJS(["M139.dom","M139.text"],function(){});
112     */
113     requireJS: function (usingNSList, callback) {
114         var count = 0;
115         var notLoadNS = [];
116         var loadingNS = [];
117         var loadedNS = [];
118         for (var i = 0; i < usingNSList.length; i++) {
119             var ns = usingNSList[i];
120             var readyStatus = this.checkNSLoaded(ns);
121             switch (readyStatus) {
122                 case this.NSLOADEDSTATUS.NOTLOAD:
123                     notLoadNS.push(ns);
124                     break;
125                 case this.NSLOADEDSTATUS.LOADING:
126                     loadingNS.push(ns);
127                     break;
128                 case this.NSLOADEDSTATUS.LOADED:
129                     loadedNS.push(ns);
130                     break;
131                 default:
132                     throw this.THROWS.NSNOTEXIST + ns;
133                     break;
134             }
135         }
136         count = loadedNS.length;
137         var notReadyNS = loadingNS.concat(notLoadNS);
138         //所有ns都已经加载过了
139         if (count === usingNSList.length) {
140             doCallback();
141         } else {
142             this.on("namespaceload", function (e) {
143                 if ($.inArray(e.namespace, notReadyNS)>-1) {
144                     count++;
145                     if (count === usingNSList.length) {
146                         this.off("namespaceload", arguments.callee);
147                         doCallback();
148                     }
149                 }
150             });
151             var This = this;
152             for (var i = 0; i<notLoadNS.length; i++) {
153                 this.loadNS(notLoadNS[i], function () {
154                     This.trigger("namespaceload",{
155                         namespace:ns,
156                         path:path
157                     });
158                 });
159             }
160         }
161         function doCallback() {
162             setTimeout(callback, 0);
163         }
164     },
165     /**
166     *判断该名称空间是否已加载
167     *@returns {NSLOADEDSTATUS} 返回0:未加载，1:加载中，2:已加载
168     *@inner
169     */
170     checkNSLoaded: function (ns) {
171         if(this.get("loadingJSPath")[ns]){
172             return this.NSLOADEDSTATUS.LOADING;
173         }else if(this.get("loadedJSPath")[ns]){
174             return this.NSLOADEDSTATUS.LOADED;
175         }else if(this.get("definedJSNSPath")[ns]){
176             return this.NSLOADEDSTATUS.NOTLOAD;
177         }else{
178             var obj;
179             try{
180                 obj = eval(ns);
181             }catch(e){}
182             if(obj){
183                 return this.NSLOADEDSTATUS.LOADED;
184             }else{
185                 return -1;
186             }
187         }
188     },
189     /**@inner*/
190     loadNS:function(ns,callback){
191         var path = this.getJSPathByNS(ns);
192         this.utilCreateScriptTag({
193             id:ns,
194             src:path
195         }, function () {
196             //这里要先检查
197 
198             this.trigger("namespaceload",{
199                 namespace:ns,
200                 path:path,
201             });
202             delete this.get("loadingJSPath")[ns];
203             this.get("loadedJSPath")[ns] = 1;
204         });
205         this.get("loadingJSPath")[ns] = 1;
206     },
207     /**
208     *动态加载script标签
209     *@param {Object} options 配置
210     *@param {Stirng} options.id script标签的id ; 
211     *@param {Stirng} options.src JS文件地址（完整路径）; 
212     *@param {Stirng} options.charset 给script标签加charset属性
213     *@param {Function} callback 加载完成的回调
214     *@example
215     *M139.core.utilCreateScriptTag(
216          {
217             id:"examplejs",
218             src:"http://images.139cm.com/m2012/richmail/js/example.js",
219             charset:"utf-8"
220          },
221          function(){
222              alert("文件加载完毕");
223          }
224     *);
225     */
226     utilCreateScriptTag: function (options, callback) {
227         var This = this;
228         if(callback){
229             var _callback = callback;
230             var callback = function(){
231                 _callback.call(This);
232             }
233         }
234         var scriptId = options.id;
235         var dataHref = options.src;
236         var charset = options.charset;
237         var isReady = false;
238         var head = document.getElementsByTagName("head")[0];
239         var objScript = document.getElementById(scriptId);
240         //是否移出脚本DOM(非IE9时处理)
241         var isRemoveScriptDom = !document.all && true || false,
242 		browserVersion = ["msie 9.0", "chrome", "firefox"],
243 		i = 0, bvLenght = browserVersion.length - 1
244         currVersion = window.navigator.userAgent.toLowerCase() || "";
245         //IE9、chrome、firefox时处理
246         while (i <= bvLenght) {
247             isRemoveScriptDom = currVersion.indexOf(browserVersion[i]) > -1 && true || false;
248             if (isRemoveScriptDom) {
249                 break;
250             }
251             i++;
252         }
253         browserVersion = null;
254         try {
255             if (objScript && isRemoveScriptDom) {
256                 objScript.src = "";
257                 objScript.parentNode.removeChild(objScript);
258                 objScript = null;
259             }
260         }
261         catch (e) { }
262         if (objScript != null) {
263             if (dataHref.indexOf("?") == -1) dataHref += "?";
264             dataHref += "&" + Math.random();
265             objScript.src = dataHref;
266             var dataScript = objScript;
267         } else {
268             var dataScript = document.createElement("script");
269             dataScript.id = scriptId;
270             if (charset) {
271                 dataScript.charset = charset;
272             } 
273             dataScript.src = dataHref;
274             dataScript.defer = true;
275             dataScript.type = "text/javascript";
276             head.appendChild(dataScript);
277         }
278         if (document.all) {
279             dataScript.onreadystatechange = function () {
280                 if (dataScript.readyState == "loaded" || dataScript.readyState == "complete") {
281                     isReady = true;
282                     if (callback) callback();
283                 }
284             }
285         } else {
286             dataScript.onload = function () {
287                 isReady = true;
288                 if (callback) callback();
289             }
290         }
291     },
292 });
