138          *注册按钮
139          *@param {Object} options 配置参数集
140          *@param {String} options.name 按钮名称，作为键值
141          *@param {String} options.template 按钮的html代码
142          *@param {String} options.command 按钮绑定的指令
143          *@param {String} options.menu 按钮绑定的菜单
144          *@param {Function} options.callback 点击按钮后的回调
145          *@param {Function} options.queryStateCallback 查询状态回调（比如当前选中的文字颜色对此按钮的表现有影响）
146          *@param {Boolean} isCommonButton 是否常用按钮(放在第一排)
147         */
148         registerButton: function (options, isCommonButton) {
149             var This = this;
150 
151             var toolBar = isCommonButton ? this.toolBar_Common : this.toolBar_More;
152 
153             if (options.isLine) {
154                 //添加分割线
155                 jQuery(options.template).appendTo(toolBar)
156             } else {
157                 //添加按钮的dom元素
158                 var btn = jQuery(options.template).appendTo(toolBar).click(function (e) {
159                     This.onButtonClick(this, e, options);
160                 }).bind("dblclick",function(e){
161                     This.onButtonDblClick(this, e, options);
162                 });
163 
164                 if (options.queryStateCallback) {
165                     //当光标选择区域变化的时候，需要通知到按钮变更外观
166                     this.editor.on("bookmarkchange", function (e) {
167                         options.queryStateCallback({
168                             selectedStyle: e.selectedStyle,
169                             editor:this,
170                             element: btn
171                         });
172                     });
173                 }
174                 if(options.init){
175                     options.init({
176                         editor:this.editor,
177                         element:btn
178                     });
179                 }
180             }
181             this.buttons[options.name] = options;
182

239         /**@inner*/
240         onButtonClick: function (button, e, buttonOptions) {
241             //点击色块，直接设置字体颜色，硬编码
242             var target = M139.Dom.findParent(e.target, "span") || e.target;
243             if (target.id == "ED_SetFontColor") {
244                 this.editor.setForeColor($(target).find("span").css("background-color"));
245                 return;
246             } else if (target.id == "ED_SetBackgroundColor") {
247                 this.editor.setBackgroundColor($(target).find("span").css("background-color"));
248                 return;
249             }
250 
251             if (buttonOptions.menu) {
252                 this.showMenu({
253                     name: buttonOptions.menu,
254                     dockElement: button
255                 });
256             }
257             if (buttonOptions.command) {
258                 this.editor[buttonOptions.command]();
259             }
260 
261 
262             var btn = M139.Dom.findParent(e.target,"a");
263             var command = "";
264             if(btn.id){
265                 command = btn.id.replace("ED_","");
266             }
267 
268             this.trigger("buttonclick", {
269                 event: e,
270                 command:command,
271                 target: button,
272                 options: buttonOptions
273             });
274         },
