 10     /**@lends M2012.UI.PageTurning.prototype*/
 11     {
 12        /** 弹出菜单组件
 13         *@constructs M2012.UI.PageTurning
 14         *@extends M139.View.ViewBase
 15         *@param {Object} options 初始化参数集
 16         *@param {Number} options.pageCount 页数
 17         *@param {Number} options.pageIndex 初始化下标（第几页，从1开始）
 18         *@param {String} options.template 组件的html代码
 19         *@param {String} options.selectButtonTemplate 当有下拉按钮的时候，按钮的模板
 20         *@param {String} options.selectButtonInsertPath 下拉按钮插入的位置
 21         *@param {String} options.selectButtonPath 下拉按钮插入的位置
 22         *@param {String} options.nextButtonTemplate
 23         *@param {String} options.nextButtonInsertPath
 24         *@param {String} options.nextButtonPath
 25         *@param {String} options.prevButtonTemplate
 26         *@param {String} options.prevButtonInsertPath
 27         *@param {String} options.prevButtonPath
 28         *@param {String} options.pageNumberButtonTemplate
 29         *@param {String} options.pageNumberButtonInsertPath
 30         *@param {String} options.pageNumberButtonPath
 31         *@param {String} options.pageNumberContentPath
 32         *@param {String} options.maxPageButtonShow 一次最多显示几个1,2,3按钮
 33         *@example
 34         var pt = new M2012.UI.PageTurning({
 35             template:'<div class="blacklist-page"></div>',
 36             pageNumberButtonTemplate:'<a rel="number" href="javascript:;"></a>',
 37             pageNumberButtonPath:"a[rel='number']",
 38             prevButtonTemplate:'<a rel="prev" href="javascript:;">上一页</a>',
 39             prevButtonPath:"a[rel='prev']",

 42             numberButtonFocusClass:"on"
 43         });
 44         */
 45         initialize: function (options) {
 46             var $el = jQuery(options.template);
 47             this.setElement($el);
 48             return superClass.prototype.initialize.apply(this, arguments);
 49         },
 50         name: namespace,
 51         /**构建dom函数*/
 52         render:function(){
 53             var options = this.options;
 54             this.renderChildren(options);
 55             this.initEvent();
 56             return superClass.prototype.render.apply(this, arguments);
 57         },
 58         /**
 59          *创建内容html
 60          *@inner
 61          */
 62         renderChildren:function(options){
 63             this.options.pageIndex = options.pageIndex;
 64             this.options.pageCount = options.pageCount;
 65 
 66             //如果有下拉页码按钮
 67             if (options.selectButtonTemplate) {
 68                 this.$el.append(options.selectButtonTemplate);
 69                 this.$(options.pageLabelPath).text(options.pageIndex + "/" + options.pageCount);
 70             }
 71 
 72             var showPrev = options.pageIndex > 1;
 73             var btnPrev = $(options.prevButtonTemplate);
 74             //插入上一页按钮
 75             if(options.prevButtonTemplate){
 76                 if(options.prevButtonInsertPath){
 77                     btnPrev.appendTo(this.$el.find(options.prevButtonInsertPath));
 78                 }else{
 79                     btnPrev.appendTo(this.$el);
 80                 }
 81             }
 82             //如果不显示，可以消失或者灰显
 83             if (!showPrev) {
 84                 if (options.disablePrevButtonClass) {
 85                     btnPrev.addClass(options.disablePrevButtonClass);
 86                 } else {
 87                     btnPrev.hide();
 88                 }
 89             }
 90 
 91             //如果有页码按钮
 92             if (options.pageNumberButtonPath) {
 93                 //插入页码按钮
 94                 if(options.pageCount > 1){
 95                     if(options.pageNumberButtonTemplate){
 96                         var pageNumberContainer = options.pageNumberButtonInsertPath ? this.$el.find(options.pageNumberButtonInsertPath) : this.$el;
 97                         var startPage = 1;
 98                         var endPage = options.pageCount;
 99 
100                         //优化每次显示最佳的目标页码按钮
101                         if(options.maxPageButtonShow && options.maxPageButtonShow < options.pageCount){
102                             startPage = Math.max(options.pageIndex-2,1);
103                             startPage = Math.min(startPage,options.pageCount - options.maxPageButtonShow + 1);
104                             endPage = Math.min(startPage + options.maxPageButtonShow - 1,options.pageCount);
105                         }
106                     
107                         for(var i=startPage;i<=endPage;i++){
108                             var numberButton = jQuery(options.pageNumberButtonTemplate).appendTo(pageNumberContainer);
109                             if(options.pageNumberContentPath){
110                                 numberButton.find(options.pageNumberContentPath).text(i);
111                             }else{
112                                 numberButton.text(i);
113                             }
114                             numberButton.attr("index",i);
115                         }
116                     }
117                 }
118 
119                 //当前页码获得焦点
120                 if(options.numberButtonFocusClass){
121                     this.$el.find("*[index='" + options.pageIndex + "']").addClass(options.numberButtonFocusClass);
122                 }
123             }
124 
125             var showNext = options.pageCount > 1 && options.pageIndex < options.pageCount;
126             var btnNext = $(options.nextButtonTemplate);
127             //插入下一页按钮
128             if(options.nextButtonTemplate){
129                 if(options.nextButtonInsertPath){
130                     btnNext.appendTo(this.$el.find(options.nextButtonInsertPath));
131                 } else {
132                     btnNext.appendTo(this.$el);
133                 }
134             }
135             if (!showNext) {
136                 if (options.disableNextButtonClass) {
137                     btnNext.addClass(options.disableNextButtonClass);
138                 } else {
139                     btnNext.hide();
140                 }
141             }
142         },
143         /**
144          *绑定事件
145          *@inner
146          */
147         initEvent:function(){
148             var This = this;
149             var options = this.options;
150             if(options.pageNumberButtonPath){
151                 this.$el.find(options.pageNumberButtonPath).click(function(){
152                     var index = this.getAttribute("index") * 1;
153                     This.onNumberButtonClick(index);
154                 });
155             }
156 
157             if(options.prevButtonPath){
158                 this.$el.find(options.prevButtonPath).click(function(){
159                     This.onPrevButtonClick();
160                 });
161             }
162 
163             if(options.nextButtonPath){
164                 this.$el.find(options.nextButtonPath).click(function(){
165                     This.onNextButtonClick();
166                 });
167             }
168 
169             if (options.selectButtonPath) {
170                 this.$(options.selectButtonPath).click(function () {
171                     This.onSelectPageClick();
172                 });
173             }
174         },
175 
176         /**
177          *程序控制强制翻页
178          *@param {Number} pageIndex 更新当前页码
179          *@param {Number} pageCount 可选参数，更新页数，
180          *@param {Boolean} isSilent 可选参数，是否静默（即不触发pagechange），默认值为false
181          *@example
182          var pt = M2012.UI.PageTurning.create({
183             pageIndex:1
184             pageCount:20,
185             maxPageButtonShow:5,
186             container:$("#ddd")
187          });
188          pt.on("pagechange",function(pageIndex){
189             console.log("点击了第"+pageIndex+"页");
190          });
191          pt.update(2);//翻到第二页
192          */
193         update: function (pageIndex, pageCount, isSilent) {
194             this.$el.html("");
195 
196             pageCount = typeof(pageCount) == "number" ? pageCount : this.options.pageCount;
197 
198             var newOp = _.defaults({
199                 pageIndex: pageIndex,
200                 pageCount: pageCount
201             }, this.options);
202 
203             this.renderChildren(newOp);
204             this.initEvent();
205 
206             if(arguments.length == 2 && _.isBoolean(arguments[1])){
207                 isSilent = arguments[1];
208             }
209 
210             if (!isSilent) {
211                 /**
212                 *翻页事件
213                 *@event 
214                 *@name M2012.UI.PageTurning#pagechange
215                 *@param {Number} index 页码
216                 */
217                 this.trigger("pagechange", pageIndex);
218             }
219         },
220         /**@inner*/
221         onSelectPageClick: function () {
222             var This = this;
223             //显示下拉菜单
224             var popup = M139.UI.Popup.create({
225                     target: this.$(this.options.selectButtonPath),
226                     width: 135,
227                     buttons: [{
228                         text: "确定",
229                         cssClass: "btnNormal",
230                         click: function () {
231                             var index = popup.contentElement.find("input:text").val();
232                             if (/^\d+$/.test(index)) {
233                                 if (index > 0 && index <= This.options.pageCount) {
234                                     This.onNumberButtonClick(parseInt(index));
235                                 } else {
236                                     top.$Msg.alert("页数错误，无法跳转", {
237                                         icon:"warn"
238                                     });
239                                 }
240                             }
241                             popup.close();
242                         }
243                     }],
244                     content: '<div style="padding-top:15px;">跳转到第 <input type="text" style="width:30px;"/> 页</div>'
245                 }
246 			);
247             popup.render();
248             popup.contentElement.find("input:text").keyup(function (e) {
249                 this.value = this.value.replace(/\D/g, "");
250             }).focus();
251             M139.Dom.bindAutoHide({
252                 element: popup.contentElement[0],
253                 stopEvent: true,
254                 callback: function () {
255                     popup.contentElement.remove();
256                 }
257             });
258         },
259         /**@inner*/
260         onNumberButtonClick:function(index){
261             this.update(index);
262         },
263         /**@inner*/
264         onPrevButtonClick:function(){
265             var index = this.options.pageIndex - 1;
266             if (index > 0) {
267                 this.update(index);
268             }
269         },
270         /**@inner*/
271         onNextButtonClick:function(){
272             var index = this.options.pageIndex + 1;
273             if (index <= this.options.pageCount) {
274                 this.update(index);
275             }
276         }
