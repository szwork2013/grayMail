 12         /**
 13         *老版本的代码，根据dom处理参数的上下文找到event参数，兼容多浏览器
 14         *@example
 15         document.onclick = function(e){
 16             e = M139.Event.getEvent(e);//或者 M139.Event.getEvent()
 17         }
 18         */
 19         getEvent: function (A) {
 20             var evt = A || window.event;
 21             if (!evt) {
 22                 var arr = [], C = this.getEvent.caller;
 23                 while (C) {
 24                     evt = C.arguments[0];
 25                     if (evt && (evt.constructor.target || evt.srcElement)) {
 26                         break;
 27                     }
 28                     var B = false;
 29                     for (var D = 0; D < arr.length; D++) {
 30                         if (C == arr[D]) {
 31                             B = true;
 32                             break;
 33                         }
 34                     }
 35                     if (B) {
 36                         break;
 37                     } else {
 38                         arr.push(C);
 39                     }
 40                     C = C.caller;
 41                 }
 42             }
 43             return evt;
 44         },
 45 
 46         /**
 47         *同时取消事件冒泡，以及默认行为，即：stopPropagation和preventDefault，兼容IE
 48         *@example
 49         document.onclick = function(e){
 50             M139.Event.stopEvent(e);
 51         }
 52         */
 53         stopEvent: function (e) {
 54             if (!e) {
 55                 e = this.getEvent();
 56             }
 57             if (e) {
 58                 if (e.stopPropagation) {
 59                     e.stopPropagation();
 60                     e.preventDefault();
 61                 }
 62                 else {
 63                     e.cancelBubble = true;
 64                     e.returnValue = false;
 65                 }
 66             }
 67         },
