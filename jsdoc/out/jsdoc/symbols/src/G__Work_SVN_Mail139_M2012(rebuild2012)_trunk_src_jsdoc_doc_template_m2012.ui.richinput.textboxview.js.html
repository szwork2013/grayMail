  5 (function (jQuery, _, Backbone, M139) {
  6     var namespace = "M2012.UI.RichInput.TextBoxView";
  7     M139.namespace(namespace, Backbone.View.extend(
  8         /**@lends M2012.UI.RichInput.TextBoxView.prototype*/
  9         {
 10             /** 定义通讯录地址本组件代码
 11              *@constructs M2012.UI.RichInput.TextBoxView
 12              *@param {Object} options 初始化参数集
 13              *@param {HTMLElement} options.element 托管的文本框对象
 14              */
 15             initialize: function (options) {
 16 
 17                 this.setElement(options.element);
 18 
 19                 this.richInputBox = options.richInput;
 20 
 21                 this.jTextBox = this.$("input:text");
 22                 this.textbox = this.jTextBox[0];
 23

 25             },
 26             /**
 27              *初始化事件
 28              *@inner
 29              */
 30             initEvent: function () {
 31                 this.jTextBox.click($.proxy(this, "onClick"))
 32                     .focus($.proxy(this, "onFocus"))
 33                     .blur($.proxy(this, "onBlur"))
 34                     .keydown($.proxy(this, "onKeyDown"))
 35                     .keydown($.proxy(this, "onKeyUp"))
 36                     .bind("paste", $.proxy(this, "onPaste"))
 37                     .bind("cut", $.proxy(this, "onCut"));
 38                 var This = this;
 39                 M139.Timing.watchInputChange(this.textbox, function (e) {
 40                     This.onChange(e);
 41                 });
 42             },
 43             /**
 44              *文本框内容变更时
 45              *@inner
 46              */
 47             onChange: function (e) {
 48                 this.fixTextBoxWidth();
 49                 this.trigger("input");
 50             },
 51             /**
 52              *文本框根据内容自适应宽度
 53              *@inner
 54              */
 55             fixTextBoxWidth: function () {
 56                 var jText = this.jTextBox;
 57                 var minWidth = 10;
 58                 if (jText.val() == "") {
 59                     this.$el.width(minWidth);
 60                     return;
 61                 }
 62                 if ($.browser.msie) {
 63                     var width = jText[0].createTextRange().boundingWidth + 13;
 64                 } else {
 65                     //计算宽度
 66                     var widthHelper = $("#widthHelper");
 67                     if (widthHelper.length == 0) {
 68                         widthHelper = $("<span style='position:absolute;left:0px;top:0px;visibility:hidden;'\
 69                     id='widthHelper'></span>").appendTo(document.body);
 70                         widthHelper.css({
 71                             fontSize: jText.css("font-size"),
 72                             fontFamily: jText.css("font-family"),
 73                             border: 0,
 74                             padding: 0
 75                         });
 76                     }
 77                     var width = widthHelper.text(jText.val().replace(/ /g, "1")).width() + 13;
 78                 }
 79                 var maxWidth = this.richInputBox.$el.width() - 3;
 80                 if (width > maxWidth) width = maxWidth;
 81                 if (width < minWidth) width = minWidth;
 82 
 83                 this.$el.width(width);
 84                 jText.width(width);
 85             },
 86 
 87             setEditMode: function (itemView) {
 88                 var jTextBox = this.jTextBox;
 89                 jTextBox.attr("mode", "edit"); //防止自动触发blur
 90                 setTimeout(function () {
 91                     jTextBox.attr("mode", "");
 92                 }, 0);
 93                 jTextBox.val(itemView.allText);
 94                 itemView.$el.replaceWith(this.$el);
 95                 itemView.remove();
 96                 
 97                 M139.Dom.selectTextBox(this.textbox);
 98                 this.fixTextBoxWidth();
 99             },
100 
101             setValue:function(value){
102                 this.textbox.value = value;
103                 this.fixTextBoxWidth();

166             },
167 
168             /**
169              *按键松开
170              *@inner
171              */
172             onKeyUp: function (e) {
173                 this.fixTextBoxWidth();
174             },
175             /**
176              *键盘按下
177              *@inner
178              */
179             onKeyDown: function (e) {
180                 var This = this;
181                 if (e.shiftKey || e.ctrlKey) return;
182                 var current = this.richInputBox;
183                 var Keys = M139.Event.KEYCODE;
184                 var textbox = this.textbox;
185                 this.fixTextBoxWidth();
186                 switch (e.keyCode) {
187                     case Keys.BACKSPACE:
188                         {
189                             return KeyDown_Backspace.apply(this, arguments);
190                         }
191                     case Keys.DELETE:
192                         {
193                             return KeyDown_Delete.apply(this, arguments);
194                         }
195                     case Keys.SEMICOLON:
196                     case Keys.COMMA:
197                     case Keys.ENTER:
198                         {
199                             return KeyDown_Enter.apply(this, arguments);
200                         }
201                     case Keys.LEFT:
202                         {
203                             return KeyDown_Left.apply(this, arguments);
204                         }
205                     case Keys.RIGHT:
206                         {
207                             return KeyDown_Right.apply(this, arguments);
208                         }
209                     case Keys.UP: case Keys.Down:
210                         {
211                             e.isUp = e.keyCode == Keys.Up;
212                             return KeyDown_Up_Down.apply(this, arguments);
213                         }
214                     case Keys.TAB:
215                         {
216                             return KeyDown_Tab.apply(this, arguments);
217                         }
218                 }
219                 function KeyDown_Backspace(e) {
220                     if (textbox.value == "") {
221                         if (current.getSelectedItems().length > 0) return;
222                         var item = current.getTextBoxPrevItem();
223                         if (item) item.remove();
224                         textbox.focus();
225                     }
226                 }
227                 function KeyDown_Delete(e) {
228                     if (textbox.value == "") {
229                         var item = current.getTextBoxNextItem();
230                         if (item) item.remove();
231                         textbox.focus();
232                     }
233                 }
234                 function KeyDown_Enter(e) {
235                     if (textbox.value.trim() != "") {
236                         setTimeout(function () {
237                             current.createItemFromTextBox();
238                         }, 0);
239                     }
240                     return false;
241                 }
242                 function KeyDown_Left(e) {
243                     if (textbox.value == "") {
244                         var item = current.getTextBoxPrevItem();
245                         if (item) {
246                             current.moveTextBoxTo(item.$el);
247                             return false;
248                         }
249                     }
250                 }
251                 function KeyDown_Right(e) {
252                     if (textbox.value == "") {
253                         var item = current.getTextBoxNextItem();
254                         if (item) {
255                             current.moveTextBoxTo(item.$el, true);
256                             return false;
257                         }
258                     }
259                 }
260                 function KeyDown_Up_Down(e) {
261                     if (textbox.value == "") {
262                         var offset = current.textboxView.offset();
263                         var nearItems = M2012.UI.RichInput.Tool.getNearlyElement({
264                             x: offset.left,
265                             y: offset.top + (e.isUp ? -5 : 20),
266                             richInputBox: current
267                         });
268                         if (nearItems) current.moveTextBoxTo(nearItems.element, nearItems.isAfter);
269                         return false;
270                     }
271                 }
272                 function KeyDown_Tab(e) {
273                     textbox.setAttribute("TabPress", "1");
274                     setTimeout(function () {
275                         textbox.setAttribute("TabPress", null);
276                     }, 0);
277                 }
278             }
279         }));
