 15     *登录
 16     */
 17     login:function(options,callback){
 18         var This = this;
 19         //选择环境：研发、测试线、生产线
 20         for(var i=0;i<options.length;i++){
 21             var line = options[i];
 22             if(location.host.indexOf(line.site) > -1){
 23                 options = options[i];
 24                 break;
 25             }
 26         }
 27 
 28         var successUrl = "http://" + location.host + "/m2012/js/testing/loginsuccess.html";
 29         var loginUrl = "http://" + location.host.replace(/\w+\.(mail\.10086)/,"$1")+"/login/login.ashx?";
 30         loginUrl += "&UserName=" + encodeURIComponent(options.id);
 31         loginUrl += "&Password=" + encodeURIComponent(options.password);
 32         loginUrl += "&VerifyCode=&auto=0";
 33         loginUrl += "&loginSuccessUrl=" + encodeURIComponent(successUrl);
 34 
 35         $("<iframe style='display:none' name='loginproxy' id='loginproxy'></iframe>").appendTo(document.body).load(function(){
 36             var win = this.contentWindow;
 37             var m = win.location.href.match(/sid=([^&]+)/);
 38             if(m){
 39                 var sid = m[1];
 40                 if(callback)callback();
 41                 This.trigger("login",sid);
 42             }
 43         }).attr("src",loginUrl);
 44     }
 45  });
