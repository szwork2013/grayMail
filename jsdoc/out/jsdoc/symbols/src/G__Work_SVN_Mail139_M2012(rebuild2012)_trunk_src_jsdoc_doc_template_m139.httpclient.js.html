 40         /**
 41         *AJAX方法的封装 *注：每一次ajax方法都会重新创建一个XMLHttpRequest对象，因此可以重复发请求
 42         *@param {Object} options 配置参数
 43         *@param {String} options.url 请求提交的地址
 44         *@param {String} options.method http方法，post或者get，默认是get
 45         *@param {String} options.data 发送的数据，一般为String类型，如果data为对象，则默认以url-encoding的方式转换为key=value&key1=
 46         *@param {Number} options.timeout http方法，请求超时时间设置(毫秒)，为了避免阻塞其它请求，timeout还是很有必要的
 47         *@param {Object} options.headers http头设置，比如：Content-Type
 48         *@param {Function} callback 响应回调，自动监听为 client.on("response",callback);
 49         *@returns {M139.HttpClient} 返回对象自身
 50         *@example
 51         httpClient.request(
 52             {
 53                 method:"post",
 54                 timeout:10000,
 55                 url:"/s?func=mbox:listMessage",
 56                 headers:{
 57                     "Content-Type":"text/javascript"
 58                 }
 59             },
 60             function(e){
 61                 console.log(e.status);//http返回码，200,404等
 62                 console.log(e.isTimeout);//返回是否超时
 63                 console.log(e.responseText);//http返回码，200,404等
 64                 console.log(e.getHeaders());//返回的http头集合，使用函数因为默认处理http头会消耗性能
 65             }
 66         );
 67         */
 68         request: function (options, callback) {
 69             this.requestOptions = options;
 70             var This = this;
 71             if (this.xhr) {
 72                 throw this.logger.getThrow("一个HTTPClient实例只能执行一次request操作");
 73             }
 74             //在时间中处理是否要取消当前请求（改为从代理页发起）
 75             this.trigger("beforerequest", options, callback);
 76 
 77             if (options.cancel) {
 78                 return;
 79             }
 80 
 81             var xhr = this.xhr = this.utilCreateXHR(options);
 82             if (options.timeout) {
 83                 var timer = setTimeout(function () {
 84                     if (xhr.readyState == 3 && xhr.status == 200) return;
 85                     xhr.abort();
 86                     This.onTimeout();
 87                 }, options.timeout);
 88             }
 89             if (callback) {
 90                 this.on("response", callback);
 91             }
 92             xhr.onreadystatechange = function (data) {
 93                 if (xhr.readyState == 4 && xhr.status != 0) {//abort()后xhr.status为0
 94                     clearTimeout(timer);
 95                     if (xhr.status == 304 || (xhr.status >= 200 && xhr.status < 300)) {
 96                         This.onResponse({
 97                             responseText: xhr.responseText,
 98                             status: xhr.status,
 99                             getHeaders: function () {
100                                 return This.utilGetHttpHeaders(xhr);
101                             }
102                         });
103                     } else {
104                         This.onError({
105                             status: xhr.status,
106                             responseText: xhr.responseText
107                         });
108                     }
109                 }
110             }
111             //todo 硬编码
112             if (options.url.indexOf("/g2/") > -1) {
113                 if (/cookiepartid=1(;|$)/.test(document.cookie)) {
114                     options.url = options.url.replace("/g2/", "/g3/");
115                 }
116             }
117 
118 
119             xhr.open(options.method || "get", options.url, true);
120 
121             var data = options.data;
122 
123             //如果到了这里data仍为object类型，则自动转化为urlencoded
124             if (typeof data == "object") {
125                 data = [];
126                 for (var p in options.data) {
127                     data.push(p + "=" + encodeURIComponent(options.data));
128                 }
129                 data = data.join("&");
130                 if (!options.headers) options.headers = {};
131                 if (!options.headers["Content-Type"]) {
132                     options.headers["Content-Type"] = "application/x-www-form-urlencoded";
133                 }
134             }
135 
136             this.utilSetHttpHeaders(options.headers, xhr);
137             xhr.send(data);
138             return this;
139         },
140         abort: function (options) {
141             this.xhr.bort();
142             this.onAbort();
143             return this;
144         },
145         /**@static*/
146         utilCreateXHR: function (options) {
147             //return new XMLHttpRequest();
148             var win = (options && options.window) || window;
149             if (win.XMLHttpRequest) {
150                 return new win.XMLHttpRequest();
151             }else {
152                 var MSXML = ['MSXML2.XMLHTTP.5.0', 'MSXML2.XMLHTTP.4.0', 'MSXML2.XMLHTTP.3.0', 'Microsoft.XMLHTTP'];
153                 for (var n = 0; n < MSXML.length; n++) {
154                     try {
155                         return new win.ActiveXObject(MSXML[n]);
156                         break;
157                     }
158                     catch (e) {
159                     }
160                 }
161             }
162         },
163         /**@static*/
164         utilSetHttpHeaders: function (headers, xhr) {
165             if (headers) {
166                 for (var p in headers) {
167                     xhr.setRequestHeader(p, headers[p]);
168                 }
169             }
170         },
171         /**@static*/
172         utilGetHttpHeaders: function (xhr) {
173             var lines = xhr.getAllResponseHeaders();
174             var result = {};
175             if (lines) {
176                 lines = lines.split(/\r?\n/);
177                 for (var i = 0; i < lines.length; i++) {
178                     var l = lines[i];
179                     var arr = l.split(": ");
180                     var h = arr[0].replace(/^\w|-\w/g, function (v) { return v.toUpperCase(); }); //服务端有可能返回小写的头
181                     if (h && arr[1]) {
182                         result[h] = arr[1];
183                     }
184                 }
185             }
186             return result;
187         },
188         /**@inner*/
189         onAbort: function () {
190             /**请求被中断事件
191                 * @name M139.HTTPClient#abort

262                     });

264                 }
265             }
