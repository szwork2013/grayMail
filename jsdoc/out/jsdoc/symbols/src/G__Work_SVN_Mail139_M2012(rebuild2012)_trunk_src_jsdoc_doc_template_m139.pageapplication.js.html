138         /**
139         *等待某些异步条件成熟后执行后执行操作(通常是数据加载等)，回调只执行一次
140         *@param {String} somethingReady 条件，支持分隔符“&”、“|”和“,” 一次只能用一种条件分隔符
141         *@param {Function} handler 执行回调
142         *@example
143         //当初始化数据加载、并且编辑器加载，执行回调一次
144         pageApp.await(“initdataload & editorload”,function(){});
145 
146         //当初始化数据加载以及当编辑器加载的时候，各执行回调一次
147         pageApp.await(“initdataload,editorload”,function(){});
148 
149         //当初始化数据加载 或者 编辑器加载的时候，执行回调一次
150         pageApp.await(“initdataload | editorload”,function(){});
151 
152 
153         setTimeout(function(){
154             pageApp.makeReady("initdataload",{});
155         },1000);
156 
157         setTimeout(function(){
158             pageApp.makeReady("editorload",{});
159         },2000);
160         */
161         await: function (somethingReady, handler) {
162             if (typeof somethingReady != "string") {
163                 throw this.logger.getThrow("await参数非法，必须为字符串");
164             } else if (!jQuery.isFunction(handler)) {
165                 throw this.logger.getThrow("await参数非法，必须为函数");
166             }
167             var type = 0;
168             var map = {
169                 "|": 1,
170                 ",": 2,
171                 "&": 4
172             };
173             somethingReady.replace(/&|,|\|/g, function (c) {
174                 type ^= map[c];
175             });
176 
177             if (type === map["|"] || type === map[","] || type === map["&"]) {
178                 //切割并去空格
179                 somethingReady = jQuery.map(somethingReady.split(/&|,|\|/), function (a) { return jQuery.trim(a) });
180 
181             } else if (type !== 0) {
182                 //多种操作符
183                 throw this.logger.getThrow("await参数非法，只能同时使用一种操作符");
184             }
185             this.on("somethingready", function (e) {
186                 var whatsReady = e.event;
187                 var data = e.data;
188                 if (jQuery.isArray(somethingReady)) {
189                     var index = jQuery.inArray(whatsReady, somethingReady);
190                     if (index > -1) {
191                         if (type === map["|"]) {
192                             this.off("somethingready", arguments.callee);
193                             handler.call(this, data, whatsReady);
194                         } else if (type === map[","]) {
195                             somethingReady.splice(index, 1);
196                             handler.call(this, data, whatsReady);
197                         } else if (type === map["&"]) {
198                             somethingReady.splice(index, 1);
199                             if (somethingReady.length === 0) {
200                                 this.off("somethingready", arguments.callee);
201                                 handler.call(this, data);
202                             }
203                         }
204                     }
205                 } else if (whatsReady === somethingReady) {
206                     this.off("somethingready", arguments.callee);
207                     handler.call(this, data);
208                 }
209             });
210         },

408         /**捕获客户端异常并上报日志，默认加载文件的时候就执行*/
409         utilCatchScriptError: function (win) {
410             win.onerror = window_onerror;
411             function window_onerror(msg, fileName, lineNumber) {
412                 if (typeof msg != "string") return;
413                 var stack = [];
414 
415                 var caller = arguments.callee.caller;
416                 var reg_getFunName = /function (\w*\([^(]*\))/;
417                 while (caller) {
418                     var funCode = caller.toString();
419                     var match = funCode.match(reg_getFunName);
420                     stack.push((match && match[1]) || funCode);
421                     caller = caller.caller;
422                 }
423                 var log = "ScriptError:files=" + fileName + ",lines=" + lineNumber + ",msg=" + msg;
424                 //M139.Logger.getScriptErrorLogger().error(log);
425             }
426         }
