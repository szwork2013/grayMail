286             var value = Math.max(0,x - 10);
287             value = Math.min(value,this.TotalWidth);
288             value = this.utilGetValueByPx(value);
289             if(this.isArea){
290                 var sel = this.getSelectedValue();
291                 if(value <= sel.start){
292                     this.setStartValue(value);
293                 }else if(value >= sel.end){
294                     this.setEndValue(value);
295                 }else if(value - sel.start > sel.end - value){
296                     this.setEndValue(value);
297                 }else{
298                     this.setStartValue(value);
299                 }
300             }else{
301                 this.setStartValue(value);
302             }
303         }
