  3 (function (jQuery, _, Backbone, M139) {
  4     var namespace = "M2012.UI.RichInput.DocumentView";
  5     M139.namespace(namespace, Backbone.View.extend({
  6         /** 定义通讯录地址本组件代码
  7          *@constructs M2012.UI.RichInput.DocumentView
  8          *@param {Object} options 初始化参数集
  9          *@param {HTMLElement} options.textbox 文本框对象
 10          *@example
 11          */
 12         initialize: function (options) {
 13             this.setElement(document.body);
 14 
 15             this.initEvent();
 16         },
 17         initEvent: function () {
 18             this.$el.mousemove($.proxy(this, "onMouseMove"))
 19                 .mouseup($.proxy(this, "onMouseUp"))
 20                 .mousedown($.proxy(this, "onMouseDown"));
 21         },
 22         onMouseMove: function (e) {
 23             var current = M2012.UI.RichInput.Tool.currentRichInputBox;
 24             if (!current) return;
 25             var p = {
 26                 x: e.clientX,
 27                 y: e.clientY + current.tool.getPageScrollTop(),
 28                 target: e.target
 29             };
 30             if (current.tool.dragEnable) {
 31                 current.tool.drawDragEffect(p);
 32                 current.tool.delay("drawInsertFlag", function () {
 33                     current.tool.drawInsertFlag(p);
 34                 }, 20);
 35                 e.preventDefault();
 36                 return;
 37             } else if (current.selectArea) {
 38                 //M2012.UI.RichInput.Tool.draw(current.startPosition, p);
 39                 current.trySelect(current.startPosition, p);
 40                 e.preventDefault();
 41                 return;
 42             }
 43         },
 44         onMouseDown: function (e) {
 45             var o = e.target;
 46             var current;
 47             while (o) {
 48                 //todo 巧合编程
 49                 if (o.className && o.className.indexOf("RichInputBox") > -1) {
 50                     current = M2012.UI.RichInput.getInstanceByContainer(o);
 51                     break;
 52                 }
 53                 o = o.parentNode;
 54             }
 55             M2012.UI.RichInput.Tool.currentRichInputBox = current;
 56             for (var i = 0; i < M2012.UI.RichInput.instances.length; i++) {
 57                 var item = M2012.UI.RichInput.instances[i];
 58                 if (item != current) item.unselectAllItems();
 59             }
 60         },
 61         onMouseUp: function (e) {
 62             M2012.UI.RichInput.Tool.fireDelay("drawInsertFlag");
 63             var current = M2012.UI.RichInput.Tool.currentRichInputBox;
 64             if (!current) return;
 65             if (M2012.UI.RichInput.Tool.dragEnable) {
 66                 var dragItems = M2012.UI.RichInput.Tool.dragItems;
 67                 var insertFlag = M2012.UI.RichInput.Tool.insertFlag;
 68                 if (insertFlag && dragItems && insertFlag.richInputBox) {
 69                     var insertCurrent = insertFlag.richInputBox;
 70                     for (var i = 0; i < dragItems.length; i++) {
 71                         var moveItem = dragItems[i];
 72                         insertCurrent.insertItem(moveItem.allText, {
 73                             isAfter: insertFlag.isAfter,
 74                             element: insertFlag.element,
 75                             isFocusItem: true,
 76                             testRepeat: moveItem.richInputBox == insertCurrent ? false : true //当前拖拽，不排重(手动remove)；多个实例拖拽，要排重
 77                         });
 78                         moveItem.remove();
 79                     }
 80                     for (var i = 0; i < dragItems.length; i++) {
 81                         var moveItem = dragItems[i];
 82                         moveItem.remove();
 83                     }
 84                 }
 85             } else if (current.selectArea) {
 86                 var endPosition = {
 87                     x: e.clientX,
 88                     y: e.clientY + M2012.UI.RichInput.Tool.getPageScrollTop()
 89                 };
 90                 //M2012.UI.RichInput.Tool.draw(current.startPosition, endPosition);
 91                 current.trySelect(current.startPosition, endPosition);
 92                 if (current.getSelectedItems().length == 0) {
 93                     //todo
 94                     //Utils.focusTextBox(current.textbox);
 95                 }
 96             } else {
 97                 return;
 98             }
 99             if ($.browser.msie) {
100                 //this.releaseCapture();
101                 if (M2012.UI.RichInput.Tool.captureElement) {
102                     M2012.UI.RichInput.Tool.captureElement.releaseCapture();
103                     M2012.UI.RichInput.Tool.captureElement = null;
104                 }
105             } else {
106                 window.releaseEvents(Event.MOUSEMOVE | Event.MOUSEUP);
107             }
108             M2012.UI.RichInput.Tool.dragEnable = false;
109             current.selectArea = false;
110             M2012.UI.RichInput.Tool.dragItems = null;
111             M2012.UI.RichInput.Tool.insertFlag = null;
112             M2012.UI.RichInput.Tool.hidDragEffect();
113             M2012.UI.RichInput.Tool.hidDrawInsertFlag();
114             //M2012.UI.RichInput.Tool.currentRichInputBox = null;
115         }
116     }));
117 
118     var current;
119     M2012.UI.RichInput.DocumentView.create = function () {
120         if (!current) {
121             current = new M2012.UI.RichInput.DocumentView();
122         }
123         return current;
124     }
